using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace SumaCyfr
{
    class Program
    {
        static void Main(string[] args)
        {
            uint a,Suma=0;
            while (true)
            {
                Console.WriteLine("Podaj liczbę: ");
                if (uint.TryParse(Console.ReadLine(), out a)) break;
                else Console.WriteLine("Błędna liczba! ");
            }
            while (a > 0)
            {
                Suma += a % 10;
                a /= 10;
            }
            Console.WriteLine(b);
            Console.ReadKey();
        }
    }
}
